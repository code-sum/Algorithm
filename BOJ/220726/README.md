import math     # ìµœëŒ€ê³µì•½ìˆ˜, ìµœì†Œê³µë°°ìˆ˜ ì¶œë ¥í•˜ëŠ” í•¨ìˆ˜ëŠ” math ë¼ì´ë¸ŒëŸ¬ë¦¬ ì´ìš©!

a, b = map(int, input().split())

    while b:
        a, b = b, a % b
    return a

def LCM(a, b):
    result = (a * b) // GCD(a, b)
    return result
print(LCM(a, b))
