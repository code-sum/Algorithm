import sys

    "L": (0, -1),
    "B": (1, 0),

        continue
    king_row += input_row
    king_col += input_col
    if (king_row, king_col) == (stone_row, stone_col):
        if not (0 <= stone_row+input_row < 8 and 0 <= stone_col+input_col < 8):
        	# í‚¹ì˜ ì›€ì§ì„ ë˜ëŒë¦¬ê¸°
            king_row -= input_row
            king_col -= input_col
